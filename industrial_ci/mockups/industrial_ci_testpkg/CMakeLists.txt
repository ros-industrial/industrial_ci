cmake_minimum_required(VERSION 2.8.3)
if(POLICY CMP0048)
    cmake_policy(SET CMP0048 NEW)
endif()

endif()

install(DIRECTORY test DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION})
